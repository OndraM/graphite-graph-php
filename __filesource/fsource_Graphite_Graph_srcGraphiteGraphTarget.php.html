  220.   /**
  221.    * Find the canonical name for a target.
  222.    *
  223.    * The value may be an alias or it may differ in case from the true
  224.    * target name.
  225.    *
  226.    * @param string $name Target to lookup
  227.    * @return string Proper name of target or false if not found
  228.    */
  229.   static public function canonicalName ($name) {
  230.     static $lookupMap;
  231.     if (null == $lookupMap) {
  232.       // lazily construct the lookup map
  233.       $tmp = array();
  234.       foreach (self::$validFunctions as $func => $conf) {
  235.         $tmp[mb_strtolower($func)] = $func;
  236.       }
  237.       foreach (self::$functionAliases as $alias => $func) {
  238.         $tmp[mb_strtolower($alias)] = $func;
  239.       }
  240.       $lookupMap = $tmp;
  241.     }
  242.  
  243.     // convert to lowercase and strip "delimiter" characters
  244.     $name = strtr(mb_strtolower($name), '_.-', '');
  245.     if (array_key_exists($name, $lookupMap)) {
  246.       return $lookupMap[$name];
  247.     } else {
  248.       return false;
  249.     }
  250.   } //end canonicalName
