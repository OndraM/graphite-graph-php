  172.     // find functions named in the conf data

  188.     foreach ($funcs as $name => $args) {

  190.  
  191.       if (is_scalar($args)) {

  194.  
  195.       if ($isAlias && $haveAlias) {

  244.         } //end foreach $spec

  247.       } //end if/else
  248.  
  249.       // keep track of alias application

  252.  
  253.     return $target;

  257.   /**
  258.    * Find the canonical name for a target.
  259.    *
  260.    * The value may be an alias or it may differ in case from the true
  261.    * target name.

  263.    * @param string $name Target to lookup
  264.    * @return string Proper name of target or false if not found

  266.   static public function canonicalName ($name) {
  267.     static $lookupMap;
  268.     if (null == $lookupMap) {
  269.       // lazily construct the lookup map
  270.       $tmp = array();
  271.       foreach (self::$validFunctions as $func => $conf) {
  272.         $tmp[mb_strtolower($func)] = $func;
  273.       }
  274.       foreach (self::$functionAliases as $alias => $func) {
  275.         $tmp[mb_strtolower($alias)] = $func;
  276.       }
  277.       $lookupMap = $tmp;
  278.     }
  279.  
  280.     // convert to lowercase and strip "delimiter" characters
  281.     $name = strtr(mb_strtolower($name), '_.-', '');
  282.     if (array_key_exists($name, $lookupMap)) {
  283.       return $lookupMap[$name];
  284.     } else {
  285.       return false;
  286.     }
  287.   } //end canonicalName
