  198.   public function __get ($name) {
  199.     if ('qs' == $name || 'url' == $name) {
  200.       return $this->qs();
  201.     }
  202.  
  203.     $cname = self::canonicalParamName($name);
  204.     if (array_key_exists($cname, $this->settings)) {
  205.       $val = $this->settings[$cname];
  206.  
  207.       if ('hide' == mb_substr($cname, 0, 4) &&
  208.           'hide' != mb_substr(mb_strtolower($name), 0, 4)) {
  209.         // invert inverted alias
  210.         $val = !$val;
  211.       }
  212.  
  213.       return $val;
  214.     }
  215.  
  216.     if (false === $cname) {
  217.       // invalid request
  218.       $trace = debug_backtrace();
  219.       trigger_error("Undefined property via __get(): {$name} in " .
  220.           "{$trace[0]['file']} on line {$trace[0]['line']}", E_USER_NOTICE);
  221.     }

  223.   } //end __get

  278.   /**

  283.   public function prefix ($prefix) {
  284.     if ('.' !== mb_substr($prefix, -1)) {
  285.       // ensure that prefix ends with period
  286.       // XXX: are we sure this is a good idea?
  287.       $prefix = "{$prefix}.";
  288.     }
  289.     if ('^' == $prefix[0]) {
  290.       // this is a rooted prefix, don't concat with current
  291.       $prefix = mb_substr($prefix, 1);
  292.     } else {

  366.   /**
  367.    * Add forecast, confidence bands, aberrations and metrics using the

  375.     if (!isset($opts['series'])) {
  376.       throw new Graphite_ConfigurationException(
  377.         "'series' is required for a Holt-Winters Confidence forecast");
  378.     }
  379.     $opts['series'] = $opts['series'];
  380.  
  381.     if (!isset($opts['alias'])) {
  382.       $opts['alias'] = ucfirst($name);
  383.     }
  384.  
  385.     if (!isset($opts['forecast_line']) || $opts['forecast_line']) {
  386.       $args = $opts;
  387.       $args['series'] = "holtWintersForecast({$args['series']})";
  388.       $args['alias'] = "{$args['alias']} Forecast";
  389.       $args['color'] = (isset($args['forecast_color']))?
  390.         $args['forecast_color']: 'blue';

  393.  
  394.     if (!isset($opts['bands_line']) || $opts['bands_line']) {

  397.       $args['alias'] = "{$args['alias']} Confidence";
  398.       $args['color'] = (isset($args['bands_color']))?
  399.         $args['bands_color']: 'grey';
  400.       $args['dashed'] = true;
  401.       $this->metric("{$name}_bands", $args);
  402.     }
  403.  
  404.     if (!isset($opts['aberration_line']) || $opts['aberration_line']) {
  405.       $args = $opts;
  406.       $args['series'] = "holtWintersConfidenceAbberation(keepLastValue(" .
  407.         $args['series'] . "))";
  408.       $args['alias'] = "{$args['alias']} Aberration";
  409.       $args['color'] = (isset($args['aberration_color']))?
  410.         $args['aberration_color']: 'orange';
  411.       if (isset($args['aberration_second_y']) && $args['aberration_second_y']) {
  412.         $args['second_y_axis'] = true;
  413.       }
  414.       $this->metric("{$name}_aberration", $args);
  415.     }
  416.  
  417.     if (isset($opts['critical'])) {
  418.       $criticals = $opts['critical'];
  419.       if (!is_array($criticals)) {
  420.         $criticals = explode(',', $criticals);
  421.       }
  422.       foreach ($criticals as $value) {

  424.           $opts['critical_color']: 'red';
  425.         $caption = "{$opts['alias']} Critical";
  426.         $this->line(array(
  427.           'value' => $value,
  428.           'alias' => $caption,
  429.           'color' => $color,
  430.           'dashed' => true,
  431.         ));
  432.       }
  433.     }
  434.  
  435.     if (isset($opts['warning'])) {
  436.       $warnings = $opts['warning'];
  437.       if (!is_array($warnings)) {
  438.         $warnings = explode(',', $warnings);
  439.       }
  440.       foreach ($warnings as $value) {
  441.         $color = (isset($opts['warning_color']))?
  442.           $opts['warning_color']: 'orange';
  443.         $caption = "{$opts['alias']} Warning";
  444.         $this->line(array(
  445.           'value' => $value,
  446.           'alias' => $caption,
  447.           'color' => $color,
  448.           'dashed' => true,
  449.         ));
  450.       }
  451.     }
  452.  
  453.     if (!isset($opts['color'])) {
  454.       $opts['color'] = 'yellow';
  455.     }
  456.  
  457.     if (!isset($opts['actual_line']) || $opts['actual_line']) {
  458.       $this->metric($name, $opts);
  459.     }
  460.     return $this;
  461.   } //end forecast

  464.   /**
  465.    * Generate a graphite graph description query string.
  466.    * @param string $format Format to export data in (null for graph)
  467.    * @return string Query string to append to graphite url to render this
  468.    *     graph
  469.    * @throws Graphite_ConfigurationException If required data is missing
  470.    */
  471.   public function qs ($format=null) {
  472.     $parms = array();
  473.  
  474.     foreach ($this->settings as $name => $value) {
  475.       $parms[] = self::qsEncode($name) . '=' . self::qsEncode($value);
  476.     }
  477.  
  478.     foreach ($this->targets as $target) {
  479.       $parms[] = 'target=' .
  480.         self::qsEncode(Graphite_TargetBuilder::generateTarget($target));
  481.     } //end foreach
  482.  
  483.     if (null !== $format) {
  484.       $parms[] = 'format=' . self::qsEncode($format);
  485.     }
  486.  
  487.     return implode('&', $parms);
  488.   } //end qs

  507.    */
  508.   public function ini ($file, $vars=null) {
  509.     $global = array();
  510.     $prefixes = array();
  511.     $metrics = array();
  512.  
  513.     $ini = Graphite_IniParser::parse($file, $vars);
  514.  
  515.     foreach ($ini as $key => $value) {
  516.       if (is_array($value)) {
  517.         // sub-arrays either describe prefixes or metrics
  518.         if (isset($value[':is_prefix'])) {
  519.           $prefixes[$key] = $value;
  520.  
  521.         } else {
  522.           $metrics[$key] = $value;
  523.         }
  524.  
  525.       } else {
  526.         // must be a general setting
  527.         $global[$key] = $value;
  528.       }
  529.     }
  530.     unset($ini);
  531.  
  532.     // resolve all prefixes we found
  533.     foreach ($prefixes as $name => $conf) {
  534.       $prefixes[$name] = self::resolvePrefix($conf, $prefixes);
  535.     }
  536.  
  537.     // apply global settings
  538.     foreach ($global as $setting => $args) {
  539.       $this->$setting($args);
  540.     }
  541.  
  542.     // add metrics
  543.     foreach ($metrics as $name => $conf) {
  544.       // TODO: add support for preconfigured "types" like line, forecast, etc
  545.  
  546.       if (isset($conf[':prefix'])) {
  547.         $this->prefix($prefixes[$conf[':prefix']]);
  548.       }
  549.  
  550.       // metric name is either given explicitly or inferred from section label
  551.       $metricName = (isset($conf['metric']))? $conf['metric']: $name;
  552.  
  553.       $this->metric($metricName, $conf);
  554.  
  555.       if (isset($conf[':prefix'])) {
  556.         $this->endPrefix();
  557.       }
  558.     } //end foreach
  559.  
  560.   } //end ini

  563.   /**
  564.    * Find the canonical name for a parameter.
  565.    *
  566.    * The value may be an alias or it may differ in case from the true
  567.    * parameter name.
  568.    *
  569.    * @param string $name Parameter to lookup
  570.    * @return string Proper name of parameter or false if not found
  571.    */
  572.   static public function canonicalParamName ($name) {
  573.     static $lookupMap;
  574.     if (null == $lookupMap) {
  575.       // lazily construct the lookup map
  576.       $tmp = array();
  577.       foreach (self::$validParams as $param) {
  578.         $tmp[mb_strtolower($param)] = $param;
  579.       }
  580.       foreach (self::$paramAliases as $alias => $param) {
  581.         $tmp[mb_strtolower($alias)] = $param;
  582.       }
  583.       $lookupMap = $tmp;
  584.     }
  585.  
  586.     // convert to lowercase and strip "delimiter" characters
  587.     $name = strtr(mb_strtolower($name), '_.-', '');
  588.     if (array_key_exists($name, $lookupMap)) {
  589.       return $lookupMap[$name];
  590.     } else {
  591.       return false;
  592.     }
  593.   } //end canonicalParamName

  595.  
  596.   /**
  597.    * Create a fully qualified prefix for the given configuration.
  598.    *
  599.    * @param mixed $conf Literal prefix or array of config data
  600.    * @return string Literal prefix
  601.    */
  602.   static public function resolvePrefix ($conf, $prefixes) {
  603.     if (is_array($conf)) {
  604.       $prefix = $conf['prefix'];
  605.       if (isset($conf[':prefix'])) {
  606.         // find our parent prefix
  607.         $mom = self::resolvePrefix($prefixes[$conf[':prefix']], $prefixes);
  608.         $prefix = "{$mom}.{$prefix}";
  609.       }
  610.       $conf = $prefix;
  611.     }
  612.  
  613.     return $conf;
  614.   } //end resolvePrefix

  617.   /**
  618.    * Query string specific uri encoding.
  619.    *
  620.    * Per RFC-3986:
  621.    *   URI producing applications should percent-encode data octets that
  622.    *   correspond to characters in the reserved set unless these characters
  623.    *   are specifically allowed by the URI scheme to represent data in that
  624.    *   component.
  625.    *
  626.    * Php's builtin urlencode function is a general purpose encoder. This means
  627.    * that it takes the most conservative approach to encoding. This means
  628.    * percent-encoding all octets that are not in the "unreserved" set
  629.    * (ALPHA / DIGIT / "-" / "." / "_" / "~"). Actually it goes further than
  630.    * this and encodes the tilde as well for no apparent reason other than
  631.    * potential binary compatibility with the output of early non-conforming
  632.    * user-agents.
  633.    *
  634.    * Within the "query" section of a URI there is a broader set of valid
  635.    * characters allowed without percent-encoding:
  636.    * - query         = *( pchar / "/" / "?" )
  637.    * - pchar         = unreserved / pct-encoded / sub-delims / ":" / "@"
  638.    * - unreserved    = ALPHA / DIGIT / "-" / "." / "_" / "~"
  639.    * - pct-encoded   = "%" HEXDIG HEXDIG
  640.    * - sub-delims    = "!" / "$" / "&" / "'" / "(" / ")" /
  641.    *                   "*" / "+" / "," / ";" / "="
  642.    *
  643.    * This encoder will let php do the heavy lifting with urlencode() but will
  644.    * then decode _most_ query allowed characters. We will leave "&", "=" and
  645.    * ";" percent-encoded to preserve delimiters used in the
  646.    * application/x-www-form-urlencoded encoding.
  647.    *
  648.    * @param string $str String to encode for embedding in the query component
  649.    *     of a URI.
  650.    * @return string RFC-3986 conforming encoded string
  651.    * @see RFC-3986
  652.    * @see RFC-1738
  653.    * @see HTML 4.01 Specification
  654.    */
  655.   static public function qsEncode ($str) {
  656.     static $decode = array(
  657.       '%21' => '!',
  658.       '%24' => '$',
  659.       '%27' => '\'',
  660.       '%28' => '(',
  661.       '%29' => ')',
  662.       '%2A' => '*',
  663.       '%2B' => '+',
  664.       '%2C' => ',',
  665.       '%2F' => '/',
  666.       '%3A' => ':',
  667.       '%3F' => '?',
  668.       '%40' => '@',
  669.       '%7E' => '~',
  670.     );
  671.  
  672.     $full = urlencode($str);
  673.     return str_replace(array_keys($decode), array_values($decode), $full);
  674.   } //end qsEncode
