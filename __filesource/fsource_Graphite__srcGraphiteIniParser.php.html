  9. /**
  10.  * Ini file parser.
  11.  *
  12.  * Utility class which performs Mustache-like template expansion on a file
  13.  * before parsing it into a PHP array structure.
  14.  *
  15.  * Tokens in the ini file are strings that start with `{{` and end wtih `}}`.
  16.  * The exact text inside the delimiters becomes the token. These tokens can
  17.  * be replaced with alternate content by passing an array of replacement
  18.  * values to Graphite_IniParser::parse(). The array should use the exact token
  19.  * text as the key and the replacement value as the value for each substition
  20.  * to be made. Any tokens which do not appear as keys in the replacement
  21.  * variables array will be used verbatium in the output. This allows using a
  22.  * default value as the token itself.
  23.  *
  24.  * PHP's parse_ini_string()/parse_ini_file() functions are used to parse the
  25.  * ini file. There are several things to be aware of as a result of this.
  26.  * - Semicolons (;) should be used for comments.
  27.  * - Unquoted numerics are parsed as php integers thus numbers starting by 0
  28.  *   are evaluated as octals and numbers starting by 0x are evaluated as
  29.  *   hexadecimals.
  30.  * - If a value in the ini file contains any non-alphanumeric characters it
  31.  *   needs to be enclosed in double-quotes (").
  32.  * - Values enclosed in double quotes can contain new lines.
  33.  * - Values not enclosed in double quotes are subject to constant expansion.
  34.  *   This means that values matching the names of constants defined at the time
  35.  *   the ini file is parsed will be replaced with the value of the constant.
  36.  * - There are reserved words which must not be used as keys for ini files.
  37.  *   These include: null, yes, no, true, false, on, off, none. Values null, no
  38.  *   and false results in "", yes and true results in "1". Characters
  39.  *   ?{}|&~![()^" must not be used anywhere in the key and have a special
  40.  *   meaning in the value.
  41.  *
  42.  * Example:
  43.  * <code>
  44.  * ; example ini file
  45.  * [section]
  46.  * key1 = "{{TOKEN1}}"
  47.  * key2 = "{{TOKEN2}}"
  48.  * </code>
  49.  *
  50.  * <code>
  51.  * <?php
  52.  * $ini = Graphite_IniParser::parse('/path/to/file.ini', array(
  53.  *    'TOKEN1' => 'replacement 1',
  54.  *    'TOKEN2' => 'replacement 2',
  55.  *  ));
  56.  * </code>
  57.  *
  58.  * @package Graphite
  59.  * @author Bryan Davis <bd808@bd808.com>
  60.  * @copyright 2012 Bryan Davis and contributors. All Rights Reserved.
  61.  * @license http://www.opensource.org/licenses/BSD-2-Clause Simplified BSD License
  62.  * @see parse_ini_string
  63.  * @see parse_ini_file
  64.  */

  71.   const RE_PARSE = '/
  72.     \\\\{{          # backslash escaped start marker
  73.     |               # OR
  74.     \\\\}}          # backslash escaped end marker
  75.     |               # OR
  76.     (?<!\\\\)       # not after a backslash
  77.     {{              # token start marker
  78.     (?P<label>      # start capture "label"
  79.       [^{]          # first char of token is anything but a brace
  80.       .*?           # non-greedy match of anything
  81.     )               # end capture "label"
  82.     (?<!\\\\)       # not after a backslash
  83.     }}              # token end marker
  84.     /sxS';          // dotall, extended, analyze

  115.   private function expand () {
  116.     $expanded = preg_replace_callback(

  118.  
  119.     if (function_exists('parse_ini_string')) {
  120.       return parse_ini_string($expanded, true);

  127.       unlink($tmp);

  133.   /**
  134.    * Substitute a match with data found in our variables.

  136.    * @return string Variable value or original key if no match found
  137.    */
  138.   public function substitute ($match) {
  139.     if ('\\{{' == $match[0] || '\\}}' == $match[0]) {
  140.       // strip escape char
  141.       return mb_substr($match[0], 1);
  142.     }
  143.  
  144.     $label = $match['label'];
  145.     $label = strtr($label, array('\\{{' => '{{', '\\}}' => '}}'));
  146.     if (isset($this->vars[$label])) {
  147.       return $this->vars[$label];
  148.  
  149.     } else {
  150.       return $label;
  151.     }
  152.   } //end substitute
  153.

  155.   /**
  156.    * Parse an ini file and return an array of it's contents.
  157.    *
  158.    * @param string $file Path to ini file
  159.    * @param array $vars Variable values to substiute in the file
  160.    * @return array Parsed ini data
  161.    */
  162.   static public function parse ($file, $vars=null) {
  163.     if (null === $vars) {
  164.       return parse_ini_file($file, true);
  165.     }
  166.  
  167.     $p = new Graphite_IniParser($file, $vars);
  168.     return $p->expand();
  169.   } //end parse
  170.  
  171. } //end Graphite_IniParser
